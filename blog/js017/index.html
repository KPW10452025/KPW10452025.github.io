JS 課程017 自增與自減

By Webb 2022-03-02

本篇要介紹的是，自增 ++ 與自減 –

  • 自增 ++

通過自增可以使變量在自身的基礎上增加 1

對一個變量自增以後，原變量的值會立即自增 1

var a = 1;
a = a + 1;
console.log("a =", a) // a = 2

自增分成兩種：變數++ 與 ++變數，舉例：a++ 與 ++a

無論是 a++ 還是 ++a 都會使原變量的值自增 1

var a = 1;
a++;
console.log("a =", a) // a = 2

var a = 1;
++a;
console.log("a =", a) // a = 2

注意：a++ 與 ++a 所代表的值是不同的

a++ 表示自增前的值

++a 表示自增後的值

var a = 3;
console.log("a++ =", a++) // a++ = 3
console.log("a =", a) // a = 4

var a = 3;
console.log("++a =", ++a) // ++a = 4
console.log("a =", a) // a = 4

var a = 20;
var result = a++ + ++a + a;
console.log("result =", result); // result = 64
// 20 + 22 + 22

var a = 30;
a = a++
console.log("a =", a) // 30

var a = 30;
a = ++a
console.log("a =", a) // 31
  • 自減 --

通過自減可以使變量在自身的基礎上減 1

自減也分兩種：變數-- 與 --變數，舉例：b-- 與 --b

無論是b-- 或 --b 都會使原變量自減 1

var b = 10;
b--
console.log("b =", b) //b = 9

var b = 10;
--b
console.log("b =", b) //b = 9

注意：b-- 與 --b 所代表的值是不同的

b-- 表示自減前的值

--b 表示自減後的值

var b = 3;
console.log("b-- =", b--) // b-- = 3
console.log("b =", b) // b = 2

var b = 3;
console.log("--b =", --b) // --b = 2
console.log("b =", b) // a = 2

var b = 20;
var result = b-- + --b + b;
console.log("result =", result); // result = 56
// 20 + 18 + 18
